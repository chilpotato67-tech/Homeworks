 name: My first play
  hosts: all
  become: true
  tasks:
    - name: Print a message
      ansible.builtin.debug:
        msg: "Hello world, Ansible is running!"
    - name: update ubuntu
      ansible.builtin.apt:
        update_cache: yes
        upgrade: dist 
    - name: install fontconfig
      ansible.builtin.apt:
        name: fontconfig
        state: present
          
    - name: install java
      ansible.builtin.apt:
        name: openjdk-11-jre
        state: present  
    - name: add jenkins repository key
      ansible.builtin.apt_key:
        url: https://pkg.jenkins.io/debian/jenkins.io-2026.key
        state: present      
    - name: add jenkins repository
      ansible.builtin.shell: echo "deb [signed-by=/etc/apt/keyrings/jenkins-keyring.asc]" https://pkg.jenkins.io/debian binary/ | sudo tee /etc/apt/sources.list.d/jenkins.list > /dev/null
    - name: install jenkins
      ansible.builtin.apt:
        name: jenkins
        state: present
    - name: install nginx
      ansible.builtin.apt:
        name: nginx
        state: present  
    - name: configure nginx as reverse proxy for jenkins
      ansible.builtin.copy:
        dest: /etc/nginx/sites-available/default
        content: |
          server {
              listen 80;
              location / {
                  proxy_pass http://localhost:8080;
                  proxy_set_header Host $host;
                  proxy_set_header X-Real-IP $remote_addr;
                  proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
              }
          }
      notify: restart nginx
  
  handlers:
    - name: restart nginx
      ansible.builtin.service:
        name: nginx
        state: restarted
